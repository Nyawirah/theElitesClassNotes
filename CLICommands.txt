pwd- current directory

ls- list of files in directory

ls -l  -List of files in directory with more information

cd {folder to go to} – changes directory

cd .. - go back

clear – clearing all

mkdir {new foldername} - create directory

touch {filename} – creates new file

nano - edit file 

echo- displays info to shell

cat {filename} – read file contents

rm {filename} – remove a filename

{command} --help - displays help menu

mv {filename} --moving file/folder to another place

rm -rf {foldername} – remove directory

ls - -help – a list of information on what can be done with ls

mkdir - -help - a list of information on what can be done with mkdir

{any command} - -help -a list of information on what can be done with all commands
